#include <iostream>
#include <vector>
#include "Barn.h"
using namespace std;


Barn :: Barn(){
}

void Barn :: adoptChicken(int WT, string NM){
    myPeeps.push_back(Chicken(WT, NM));
}
void Barn :: adoptCow(int WT, string NM){
    myCattle.push_back(Cow(WT, NM));
}
void Barn :: adoptPig(int WT, string NM){
    myPigs.push_back(Pig(WT, NM));
}
void Barn :: feedChickens(){
    for(int i = 0; i < myPeeps.size(); i++){
        myPeeps[i].eat();
        myPeeps[i].gainWeight(1); // only adds 1 pound since chickens don't eat very much
    }
}
void Barn :: feedCows() {
    for (int i = 0; i < myCattle.size(); i++) {
        myCattle[i].eat();
        myCattle[i].gainWeight(8);
    }
}
void Barn :: feedPigs() {
    for (int i = 0; i < myPigs.size(); i++) {
        myPigs[i].eat();
        myPigs[i].gainWeight(10);
    }
}
